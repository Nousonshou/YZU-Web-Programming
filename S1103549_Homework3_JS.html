document.addEventListener('DOMContentLoaded', function() { // Get all the navigation links const navLinks = document.querySelectorAll('header nav ul li a'); // Add a click event listener to each link navLinks.forEach(link => { link.addEventListener('click', function(event) { event.preventDefault(); // Get the target element (the element the link scrolls to) const targetId = this.getAttribute('href'); const target = document.querySelector(targetId); // Smoothly scroll to the target element window.scroll({ behavior: 'smooth', left: 0, top: target.offsetTop }); }); }); });
